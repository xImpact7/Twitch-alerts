<script>
const alertBox = document.getElementById("alert");

function showAlert(type, user, message) {
    document.getElementById("type").innerText = type;
    document.getElementById("user").innerText = user;
    document.getElementById("message").innerText = message;

    gsap.timeline()
        .set(alertBox, {opacity: 1, y: 30})
        .to(alertBox, {y: 0, duration: 0.4, ease: "power2.out"})
        .to(alertBox, {opacity: 1, duration: 3})
        .to(alertBox, {opacity: 0, y: -20, duration: 0.4});
}

//  StreamElements Events
window.addEventListener("onEventReceived", function (obj) {
    const data = obj.detail.event;
    const listener = obj.detail.listener;

    if (listener === "follower") {
        showAlert(
            "NEUER FOLLOW",
            data.name,
            "Willkommen im Stream "
        );
    }

    if (listener === "subscriber") {
        showAlert(
            "NEUER SUB",
            data.name,
            "ist nun Teil der Gang "
        );
    }

    if (listener === "cheer") {
        showAlert(
            "BITS",
            data.name,
            "Danke für die Bits!"
        );
    }

    if (listener === "tip") {
        showAlert(
            "SPENDE",
            data.name,
            "Vielen Dank für den Support!"
        );
    }

    if (listener === "raid") {
        showAlert(
            "RAID",
            data.name,
            "Willkommen im Stream – viel Spaß!"
        );
    }
});
</script>
